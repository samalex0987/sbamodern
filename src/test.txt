<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
</head>
<body>

  <style>
    #ball{
      height: 100px;
      width: 100px;
      background-color: red;
      border-radius: 100%;
      position: absolute;
    }

    #box{
      height: 200px;
      width: 200px;
      background-color: blue;
      position: absolute;
      top: 50%;
      left: 50%;
      transform: translate(-50%,-50%);
    }
  </style>

  <div id="ball"></div>
  <div id="box"></div>

  <script>
    let ball = document.getElementById("ball");
    let box = document.getElementById("box");

    document.addEventListener("mousemove", (e) => {
      const x = e.clientX;
      const y = e.clientY;

      // Move the ball based on mouse position
      ball.style.left = `${x - ball.offsetWidth / 2}px`;
      ball.style.top = `${y - ball.offsetHeight / 2}px`;

      // Get the bounding rectangles for both the ball and the box
      let ballRect = ball.getBoundingClientRect();
      let boxRect = box.getBoundingClientRect();

      // Check if the ball is touching the box
      let isTouching = (
        ballRect.right >= boxRect.left &&
        ballRect.left <= boxRect.right &&
        ballRect.bottom >= boxRect.top &&
        ballRect.top <= boxRect.bottom
      );

      // Log the result (You can replace this with any action you want, like changing colors)
      if (isTouching) {
        console.log("The ball is touching the box!");
        ball.style.backgroundColor = "green";  // Change ball color when touching
      } else {
        console.log("The ball is NOT touching the box.");
        ball.style.backgroundColor = "red";  // Keep original ball color when not touching
      }
    });
  </script>
  
</body>
</html>
